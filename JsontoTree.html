org.springframework.boot spring-boot-starter-thymeleaf import org.thymeleaf.TemplateEngine; import org.thymeleaf.context.Context; import org.thymeleaf.templatemode.TemplateMode; import org.thymeleaf.templateresolver.FileTemplateResolver; String templatePath = "path/to/template.html"; String outputFilePath = "path/to/output.html"; // Create a template resolver FileTemplateResolver templateResolver = new FileTemplateResolver(); templateResolver.setTemplateMode(TemplateMode.HTML); templateResolver.setCharacterEncoding("UTF-8"); // Create a template engine TemplateEngine templateEngine = new TemplateEngine(); templateEngine.setTemplateResolver(templateResolver); // Create a Thymeleaf context with the dynamic variables Context context = new Context(); context.setVariable("title", "My Custom Title"); context.setVariable("message", "Hello, World!"); // Process the template with the context String output = templateEngine.process(templatePath, context); // Write the processed template to a new HTML file Files.write(Paths.get(outputFilePath), output.getBytes());
